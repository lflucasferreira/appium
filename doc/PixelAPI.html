def remove_my_cart(email)
  user = get_user(email)
  user_id = user.parsed_response['objectId']
  cart = get_cart(user_id)
  cart_list = cart.parsed_response['results']
  delete_cart(cart_list.first['objectId']) if cart_list.size.positive?
end
